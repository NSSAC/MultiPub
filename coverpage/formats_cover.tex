% ---------- Build comma-separated author list in \SelectedAuthors order ----------
% Requires:
%   \usepackage{etoolbox}
%   \input{authors_db}        % defines \AllAuthorData and stub \OneAuthor
%   \newcommand{\SelectedAuthors}{...}  % defined in main_*.tex
\newcommand{\coverauthors}{%
  \def\do##1{%
    \def\TargetID{##1}%
    % For each target ID, scan all authors and print only the match
    \renewcommand{\OneAuthor}[7]{%
      \def\CurrentID{####1}%
      \ifx\CurrentID\TargetID
        ####2, %
      \fi
    }%
    \AllAuthorData
  }%
  \expandafter\docsvlist\expandafter{\SelectedAuthors}%
}
